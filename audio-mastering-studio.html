const form = document.getElementById('masterForm'); const fileInput = document.getElementById('audioFile'); const fileMeta = document.getElementById('fileMeta'); const statusText = document.getElementById('status'); const submitButton = document.getElementById('submitButton'); const presetSelect = document.getElementById('preset'); function setStatus(message, state) { statusText.textContent = message; statusText.dataset.state = state || ''; } function formatFileSize(bytes) { if (!Number.isFinite(bytes) || bytes <= 0) return '0 B'; const units = ['B', 'KB', 'MB', 'GB']; const index = Math.min(Math.floor(Math.log(bytes) / Math.log(1024)), units.length - 1); const value = bytes / Math.pow(1024, index); return `${value.toFixed(1)} ${units[index]}`; } fileInput.addEventListener('change', () => { const file = fileInput.files[0]; if (!file) return; fileMeta.textContent = `${file.name} • ${formatFileSize(file.size)}`; }); form.addEventListener('submit', async (e) => { e.preventDefault(); const file = fileInput.files[0]; if (!file) { setStatus('No file selected', 'error'); return; } submitButton.disabled = true; setStatus('Processing...', ''); try { const formData = new FormData(); formData.append('track', file); formData.append('preset', presetSelect.value); const response = await fetch('/master', { method: 'POST', body: formData }); if (!response.ok) throw new Error('Upload failed'); const blob = await response.blob(); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'mastered.wav'; a.click(); setStatus('Download ready', 'success'); } catch (err) { setStatus(err.message, 'error'); } submitButton.disabled = false; });
